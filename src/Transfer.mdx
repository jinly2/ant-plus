import { useState } from 'react';
import { Playground, Props } from 'docz';

# Transfer 穿梭框

## 演示

```js
import { Transfer } from 'antx';
```

<Playground>
  {() => {
    const [targetKeys, onChange] = useState(['tiger']);
    return (
      <Transfer
        data={[
          { key: 'tiger', title: '小脑斧 🐅' },
          { key: 'dog', title: '小狗子 🐕' },
          { key: 'cat', title: '小喵咪 🐈' },
        ]}
        targetKeys={targetKeys}
        onChange={(newTargetKeys) => {
          onChange(newTargetKeys);
        }}
        title="宠物"
        search
        unit="只"
        searchTip="找一找"
      />
    );
  }}
</Playground>

<Props of={Transfer} isToggle />
