TreeSelect 树选择

演示

import { TreeSelect } from 'antx';
 找一找
<TreeSelect
  data={[
    {
      val: 'tiger',
      text: '小脑斧 🐅',
      children: [{ val: 'lil_tiger', text: '小小脑斧 🐯' }],
    },
    {
      val: 'dog',
      text: '小狗子 🐕',
      children: [{ val: 'lil_dog', text: '小小狗子 🐶' }],
    },
    { val: 'cat', text: '小喵咪 🐈' },
  ]}
  keys={['val', 'text']}
  onChange={value => {
    console.log(value)
  }}
  tip="找一找"
  search
  checkbox
  showType={TreeSelect.SHOW_ALL}
/>
